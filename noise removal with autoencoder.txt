import numpy as np
import librosa
import matplotlib.pyplot as plt
from tensorflow.keras.models import Model
from tensorflow.keras.layers import Input, Dense
import soundfile as sf

# Load audio
noisy, sr = librosa.load('noisy_speech.wav', sr=16000)
clean, sr = librosa.load('clean_speech.wav', sr=16000)

# Convert to spectrogram
def get_spectrogram(signal):
    S = np.abs(librosa.stft(signal))
    return S

noisy_spec = get_spectrogram(noisy)
clean_spec = get_spectrogram(clean)

# Normalize
noisy_spec = noisy_spec / np.max(noisy_spec)
clean_spec = clean_spec / np.max(clean_spec)

# Reshape for model
X = noisy_spec.T
Y = clean_spec.T

# Autoencoder Model
input_dim = X.shape[1]

input_layer = Input(shape=(input_dim,))
encoded = Dense(256, activation='relu')(input_layer)
encoded = Dense(128, activation='relu')(encoded)
decoded = Dense(256, activation='relu')(encoded)
decoded = Dense(input_dim, activation='sigmoid')(decoded)

model = Model(input_layer, decoded)
model.compile(optimizer='adam', loss='mse')

# Train model
history = model.fit(X, Y, epochs=20, batch_size=32, validation_split=0.2)

# Predict (denoise)
denoised_spec = model.predict(X)

# Convert back to waveform
denoised_spec = denoised_spec.T * np.max(clean_spec)
reconstructed = librosa.istft(denoised_spec)

# Save output
sf.write('cleaned_output.wav', reconstructed, sr)

# Plot Epoch vs Error
plt.plot(history.history['loss'], label='Training Error')
plt.plot(history.history['val_loss'], label='Validation Error')
plt.xlabel('Epochs')
plt.ylabel('Error (MSE)')
plt.title('Epoch vs Error')
plt.legend()
plt.show()
